# Step 1: Import Required Libraries

import pandas as pd
import numpy as np

from sklearn.model_selection import train_test_split
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
from sklearn.metrics import accuracy_score, confusion_matrix


# Step 2: Create Sample Dataset

data = {
    'Tweet': [
        "I love this product",
        "This is amazing",
        "Very bad service",
        "I hate this app",
        "Excellent support",
        "Worst experience ever",
        "I am very happy",
        "This is terrible",
        "Fantastic work",
        "Not good at all"
    ],
    'Sentiment': [
        "Positive",
        "Positive",
        "Negative",
        "Negative",
        "Positive",
        "Negative",
        "Positive",
        "Negative",
        "Positive",
        "Negative"
    ]
}

df = pd.DataFrame(data)

df




# Step 3: Define Features and Target

X = df['Tweet']
y = df['Sentiment']


# Step 4: Convert Text into Numerical Form

cv = CountVectorizer()

X = cv.fit_transform(X)



# Step 5: Split Dataset

X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.25, random_state=42
)



# Step 6: Apply Naive Bayes Classifier

model = MultinomialNB()

model.fit(X_train, y_train)


# Step 7: Predict Values

y_pred = model.predict(X_test)

print("Predicted Sentiments:\n", y_pred)



# Step 8: Confusion Matrix

cm = confusion_matrix(y_test, y_pred)

print("Confusion Matrix:\n", cm)


# Step 9: Accuracy Score

accuracy = accuracy_score(y_test, y_pred)

print("Accuracy:", accuracy)



# Step 10: Test New Sentence

new_tweet = ["I really love this service"]

new_data = cv.transform(new_tweet)

prediction = model.predict(new_data)

print("Predicted Sentiment:", prediction[0])
