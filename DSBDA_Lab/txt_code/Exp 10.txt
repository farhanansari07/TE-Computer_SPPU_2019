# Step 1: Import Required Libraries

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns


# Step 2: Load Iris Dataset

df = sns.load_dataset('iris')

df.head()


# Step 3: Basic Information

print("Shape of Dataset:", df.shape)
print(df.info())
print(df.describe())


# Step 4: Histogram of Sepal Length

plt.figure(figsize=(8,5))

plt.hist(df['sepal_length'], bins=10)

plt.title("Histogram of Sepal Length")
plt.xlabel("Sepal Length")
plt.ylabel("Frequency")

plt.show()



# Step 5: Histogram of Petal Length

plt.figure(figsize=(8,5))

plt.hist(df['petal_length'], bins=10)

plt.title("Histogram of Petal Length")
plt.xlabel("Petal Length")
plt.ylabel("Frequency")

plt.show()



# Step 6: Boxplot of Sepal Length

plt.figure(figsize=(8,5))

sns.boxplot(x=df['sepal_length'])

plt.title("Boxplot of Sepal Length")
plt.show()



# Step 7: Boxplot of Petal Length by Species

plt.figure(figsize=(8,5))

sns.boxplot(x='species', y='petal_length', data=df)

plt.title("Petal Length by Species")
plt.show()



# Step 8: Pairplot for Full Comparison

sns.pairplot(df, hue='species')

plt.show()
